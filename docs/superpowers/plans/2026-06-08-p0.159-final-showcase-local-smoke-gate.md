```python
def _local_showcase_smoke_capability(report: dict[str, Any]) -> dict[str, Any]:
    raw_status = str(report.get("status", "missing"))
    summary = report.get("summary") if isinstance(report.get("summary"), dict) else {}
    safety = report.get("safety") if isinstance(report.get("safety"), dict) else {}
    failed = int(summary.get("failed", 1) or 0)
    http_steps = int(summary.get("http_steps", 0) or 0)
    npc_ticks = int(summary.get("npc_ticks", 0) or 0)
    downloads = int(summary.get("downloads", 0) or 0)
    unsafe_flags = [
        key
        for key in (
            "provider_calls",
            "live_provider_calls",
            "global_mutation",
            "starts_server",
            "writes_repo_local_media",
            "provider_secrets_in_report",
            "raw_media_in_report",
            "local_paths_in_report",
            "payment_links_in_report",
        )
        if safety.get(key) is True
    ]
    has_temp_storage = safety.get("uses_temporary_storage") is True
    count_ready = http_steps >= 6 and npc_ticks >= 2 and downloads >= 3
    report_ready = raw_status == "succeeded" and failed == 0 and count_ready
    safety_ready = not unsafe_flags and has_temp_storage
    if report_ready and safety_ready:
        return _capability(
            capability_id="local_showcase_smoke",
            label="Local showcase smoke",
            status="ready",
            classification="local_showcase_smoke_ready",
            command="make local-showcase-smoke",
            detail="Local capture-to-3D-to-NPC-to-print smoke proof is ready.",
            evidence=[
                f"{report.get('kind', 'local_showcase_smoke_report')}:{raw_status}",
                f"http_steps:{http_steps}",
                f"npc_ticks:{npc_ticks}",
                f"downloads:{downloads}",
            ],
        )
    classification = "local_showcase_smoke_failed"
    if unsafe_flags or not has_temp_storage:
        classification = "local_showcase_smoke_safety_failed"
    return _capability(
        capability_id="local_showcase_smoke",
        label="Local showcase smoke",
        status="blocked",
        classification=classification,
        command="make local-showcase-smoke",
        detail="Local showcase smoke must pass before final showcase readiness.",
        evidence=[
            f"{report.get('kind', 'local_showcase_smoke_report')}:{raw_status}",
            f"http_steps:{http_steps}",
            f"npc_ticks:{npc_ticks}",
            f"downloads:{downloads}",
        ],
    )
```
