Add a test that writes blocked mobile deploy and Xcode evidence into a temporary repo fixture, builds the closure packet, and asserts:

```python
def test_final_launch_closure_packet_prefers_validated_device_child_actions(
    tmp_path: Path,
) -> None:
    repo_root = _repo_fixture(tmp_path)
    _write_mobile_deploy_preflight_evidence(
        repo_root,
        {
            "kind": "mobile_deploy_preflight_evidence_report",
            "status": "blocked",
            "first_blocker": {
                "id": "backend_base_url",
                "label": "Backend base URL",
                "status": "blocked",
                "command": "make backend-device-demo",
                "detail": "Backend /health is not reachable from iPhone.",
                "validation_command": "make mobile-deploy-preflight",
            },
            "operator_actions": [
                "start backend-device-demo before device checks: make backend-device-demo; rerun make mobile-deploy-preflight"
            ],
        },
    )
    _write_mobile_xcode_build_evidence(
        repo_root,
        {
            "kind": "mobile_xcode_build_evidence_report",
            "status": "blocked",
            "first_blocker": {
                "id": "xcode_license",
                "label": "Xcode license",
                "status": "blocked",
                "command": "make mobile-xcode-build",
                "detail": "Apple SDK license must be accepted outside Codex.",
                "validation_command": "make mobile-xcode-build-evidence",
            },
            "operator_actions": [
                "accept the Xcode license outside Codex, then rerun make mobile-xcode-build-evidence"
            ],
        },
    )

    result = build_final_launch_closure_packet_report(
        settings=Settings(),
        repo_root=repo_root,
    )

    actions = result.report["operator_actions"]
    assert (
        "start backend-device-demo before device checks: make backend-device-demo; rerun make mobile-deploy-preflight"
        in actions
    )
    assert (
        "accept the Xcode license outside Codex, then rerun make mobile-xcode-build-evidence"
        in actions
    )
    assert "make backend-device-demo" not in actions
    assert "make mobile-xcode-build" not in actions
    assert "make mobile-deploy-preflight" not in actions
```
