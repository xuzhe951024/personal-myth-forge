```python
def _device_action_bundle(configured_ios_deploy_runbook: dict[str, Any]) -> dict[str, Any]:
    source_bundle = configured_ios_deploy_runbook.get("device_action_bundle")
    if not isinstance(source_bundle, dict):
        return _missing_device_action_bundle()

    actions = [
        _device_action(action)
        for action in source_bundle.get("actions", [])
        if isinstance(action, dict)
    ]
    return {
        "id": "final_configured_preflight_device_actions",
        "label": "Final Configured Preflight Device Actions",
        "source_report": "configured_ios_deploy_runbook",
        "status": _device_action_status(str(source_bundle.get("status", "blocked"))),
        "actions": actions,
        "first_action": _device_first_action(source_bundle.get("first_action"), actions),
        "summary": _device_action_summary(actions),
        "safety": _device_action_safety(),
    }
```

Also add `_missing_device_action_bundle`, `_device_action`, `_device_next_action`, `_device_first_action`, `_device_action_summary`, `_device_action_safety`, and `_device_action_status`, matching the established copy pattern in `final_handoff_index.py` while using the P0.415 ids and labels.
