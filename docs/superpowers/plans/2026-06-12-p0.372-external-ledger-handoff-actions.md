```python
def _print_fulfillment_operator_action(report: dict[str, Any]) -> str:
    next_action = report.get("next_action")
    if isinstance(next_action, dict):
        command = str(next_action.get("command", "")).strip()
        validation = str(next_action.get("validation_command", "")).strip()
        if command and validation:
            return f"{command}; rerun {validation}"
        if command:
            return command
    return "make print-fulfillment-readiness"
```

Use it as `operator_action` for `refresh_configured_treatstock_quote`.
