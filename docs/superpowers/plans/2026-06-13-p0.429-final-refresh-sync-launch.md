- [ ] **Step 3: Implement the script sync**

Update `services/backend/scripts/write_final_local_report_refresh.sh` so accepted refresh exit codes run:

```sh
set +e
(
  cd services/backend
  uv run python -m myth_forge_api.cli final-demo-launch \
    --mode local \
    --repo-root ../.. \
    --output .local/final-demo-launch-local.json
)
sync_status=$?
set -e

if [ "$sync_status" -eq 0 ] || [ "$sync_status" -eq 2 ]; then
  printf '%s\n' "accepted final demo launch sync exit code $sync_status"
  printf '%s\n' "wrote services/backend/.local/final-demo-launch-local.json"
  exit 0
fi

printf '%s\n' "final demo launch sync failed after refresh: exit $sync_status" >&2
exit "$sync_status"
```
