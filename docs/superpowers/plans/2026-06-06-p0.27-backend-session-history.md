# P0.27 Backend Session History Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a backend-owned local myth session history contract so generated sessions and NPC ticks can be recovered by session id.

**Architecture:** Add a bounded JSON-backed `LocalMythSessionStore` under an ignored backend `.local` directory, save sessions/ticks through existing endpoints, expose read endpoints, and add Swift mobile models/client calls for the new history contract. Keep this as a local demo persistence layer, not user accounts, a production database, or autonomous background NPC memory.

**Tech Stack:** FastAPI, Pydantic, pathlib JSON file storage, pytest, Swift 6 mobile core contract tests, static HTML visual evidence.

```python
from __future__ import annotations

import json
import os
import re
from datetime import datetime, timezone
from pathlib import Path
from typing import Protocol

from myth_forge_api.domain.models import (
    SESSION_ID_PATTERN,
    MythSession,
    MythSessionHistory,
    NPCAgentTick,
)

SESSION_ID_RE = re.compile(SESSION_ID_PATTERN)
DEFAULT_MAX_TICKS = 24


class MythSessionStore(Protocol):
    def save_session(self, session: MythSession) -> MythSessionHistory:
        ...

    def get_session(self, session_id: str) -> MythSession | None:
        ...

    def get_history(self, session_id: str) -> MythSessionHistory | None:
        ...

    def append_tick(self, session: MythSession, tick: NPCAgentTick) -> MythSessionHistory:
        ...


class MythSessionStoreError(RuntimeError):
    status_code = 400


class MythSessionNotFoundError(MythSessionStoreError):
    status_code = 404


class LocalMythSessionStore:
    def __init__(self, root_dir: str | Path, max_ticks: int = DEFAULT_MAX_TICKS) -> None:
        self.root_dir = Path(root_dir)
        self.max_ticks = max_ticks

    def save_session(self, session: MythSession) -> MythSessionHistory:
        history = self.get_history(session.session_id)
        next_history = MythSessionHistory(
            session_id=session.session_id,
            session=self._sanitized_session(session),
            npc_ticks=history.npc_ticks if history else [],
            updated_at=self._now(),
        )
        self._write_history(next_history)
        return next_history

    def get_session(self, session_id: str) -> MythSession | None:
        history = self.get_history(session_id)
        return history.session if history else None

    def get_history(self, session_id: str) -> MythSessionHistory | None:
        if SESSION_ID_RE.fullmatch(session_id) is None:
            return None
        path = self._history_path(session_id)
        if path is None or not path.exists():
            return None
        return MythSessionHistory.model_validate(json.loads(path.read_text(encoding="utf-8")))

    def append_tick(self, session: MythSession, tick: NPCAgentTick) -> MythSessionHistory:
        if tick.session_id != session.session_id:
            raise MythSessionNotFoundError("NPC tick does not match myth session.")
        history = self.get_history(session.session_id) or self.save_session(session)
        ticks = self._bounded_ticks(history.npc_ticks + [self._sanitized_tick(tick)], session.session_id)
        next_history = MythSessionHistory(
            session_id=session.session_id,
            session=history.session,
            npc_ticks=ticks,
            updated_at=self._now(),
        )
        self._write_history(next_history)
        return next_history

    def _history_path(self, session_id: str) -> Path | None:
        if SESSION_ID_RE.fullmatch(session_id) is None:
            return None
        root = self.root_dir.resolve()
        path = (self.root_dir / session_id / "history.json").resolve()
        try:
            path.relative_to(root)
        except ValueError:
            return None
        return path

    def _write_history(self, history: MythSessionHistory) -> None:
        path = self._history_path(history.session_id)
        if path is None:
            raise MythSessionNotFoundError("Invalid myth session id.")
        path.parent.mkdir(parents=True, exist_ok=True)
        payload = history.model_dump(mode="json")
        tmp_path = path.with_suffix(".json.tmp")
        tmp_path.write_text(json.dumps(payload, indent=2), encoding="utf-8")
        os.replace(tmp_path, path)

    def _bounded_ticks(self, ticks: list[NPCAgentTick], session_id: str) -> list[NPCAgentTick]:
        matching = [tick for tick in ticks if tick.session_id == session_id]
        matching.sort(key=lambda tick: tick.tick_index)
        return matching[-self.max_ticks :]

    def _sanitized_session(self, session: MythSession) -> MythSession:
        if session.generated_asset.uri.startswith("data:"):
            return session.model_copy(
                update={
                    "generated_asset": session.generated_asset.model_copy(
                        update={"uri": "[redacted-provider-data-uri]"}
                    )
                }
            )
        return session

    def _sanitized_tick(self, tick: NPCAgentTick) -> NPCAgentTick:
        resolution = tick.world_resolution.model_copy(
            update={
                "visible_changes": [
                    self._sanitize_text(change) for change in tick.world_resolution.visible_changes
                ]
            }
        )
        return tick.model_copy(update={"world_resolution": resolution})

    def _sanitize_text(self, value: str) -> str:
        value = re.sub(r"data:[^\s,;]+", "[redacted]", value, flags=re.IGNORECASE)
        value = re.sub(r"Authorization\s*[=:]\s*Bearer\s+[A-Za-z0-9._:-]+", "[redacted]", value, flags=re.IGNORECASE)
        value = re.sub(r"raw=[^\s,;]+", "raw=[redacted]", value, flags=re.IGNORECASE)
        value = re.sub(r"api[_-]?key\s*[=:]\s*[^\s,;]+", "api_key=[redacted]", value, flags=re.IGNORECASE)
        return value

    def _now(self) -> str:
        return datetime.now(timezone.utc).replace(microsecond=0).isoformat()
```
