# P0.7 iOS Client Scaffold Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a locally testable Swift iOS client scaffold that exercises the P0.6 capture-to-myth backend contract without requiring global Xcode configuration changes.

**Architecture:** Add `apps/mobile/ios` as a SwiftPM package with a pure Swift core module, deterministic fixtures, and unit tests. Keep SwiftUI app source as source-only scaffold for a later full-Xcode target, while validating networking, multipart upload, capture id validation, and reducer behavior with `swift test`.

**Tech Stack:** Swift 6.3, Swift Package Manager, XCTest, Foundation `URLRequest`, FastAPI backend contract fixtures, static HTML/Browser visual evidence.

import XCTest
@testable import PersonalMythForgeMobileCore

final class ModelDecodingTests: XCTestCase {
    func testDecodesObjectCaptureFixture() throws {
        let capture = try FixtureLoader.decode(ObjectCapture.self, from: "object-capture-response")

        XCTAssertEqual(capture.captureId, "cap_ba02a3816bd145b4")
        XCTAssertEqual(capture.mediaItems.first?.uri, "local-capture://cap_ba02a3816bd145b4/media_0.jpg")
        XCTAssertEqual(capture.objectObservation.label, "old brass key")
    }

    func testDecodesMythSessionFixture() throws {
        let session = try FixtureLoader.decode(MythSession.self, from: "myth-session-response")

        XCTAssertEqual(session.status, "ready_for_review")
        XCTAssertEqual(session.objectCard.label, "old brass key")
        XCTAssertEqual(session.npcReactions.count, 3)
        XCTAssertEqual(session.worldResolution.acceptedActions.count, 2)
        XCTAssertEqual(session.printCandidate.requiresUserApproval, true)
    }
}

enum FixtureLoader {
    static func decode<T: Decodable>(_ type: T.Type, from name: String) throws -> T {
        let url = Bundle.module.url(forResource: name, withExtension: "json")!
        let data = try Data(contentsOf: url)
        return try PMFJSON.decoder.decode(T.self, from: data)
    }
}
```

import Foundation

public enum PMFJSON {
    public static let decoder: JSONDecoder = {
        let decoder = JSONDecoder()
        decoder.keyDecodingStrategy = .convertFromSnakeCase
        return decoder
    }()

    public static let encoder: JSONEncoder = {
        let encoder = JSONEncoder()
        encoder.keyEncodingStrategy = .convertToSnakeCase
        encoder.outputFormatting = [.sortedKeys]
        return encoder
    }()
}
```

```swift
func testMultipartBodyIncludesMetadataAndFileWithoutLocalPaths() throws {
    var builder = MultipartFormDataBuilder(boundary: "pmf-boundary")
    let metadata = ObjectCaptureMetadata(
        label: "old brass key",
        materials: ["metal", "brass"],
        source: "phone_capture",
        captureMode: "single_photo",
        visualNotes: "worn teeth"
    )
    try builder.appendJSONField(name: "metadata_json", value: metadata)
    builder.appendFile(
        fieldName: "files",
        filename: "media_0.jpg",
        contentType: "image/jpeg",
        data: Data("fake-jpeg".utf8)
    )

    let body = String(decoding: builder.build(), as: UTF8.self)

    XCTAssertTrue(body.contains("name=\"metadata_json\""))
    XCTAssertTrue(body.contains("\"capture_mode\":\"single_photo\""))
    XCTAssertTrue(body.contains("filename=\"media_0.jpg\""))
    XCTAssertTrue(body.contains("Content-Type: image/jpeg"))
    XCTAssertFalse(body.contains("/Users/"))
}
```

public enum ForgeFlowPhase: Equatable {
