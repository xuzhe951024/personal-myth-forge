```python
def _local_lane_handoff_actions(local_sources: list[dict[str, Any]]) -> list[str]:
    source = _source_by_id(local_sources, "final_demo_launch_local")
    if not source:
        return []
    actions = source.get("operator_actions")
    if not isinstance(actions, list):
        return []
    selected: list[str] = []
    for action in actions:
        if not isinstance(action, str):
            continue
        if not (_is_provider_handoff_action(action) or _is_print_handoff_action(action)):
            continue
        selected.append(f"final_demo_launch_local: {action.strip()}")
    return selected
```
