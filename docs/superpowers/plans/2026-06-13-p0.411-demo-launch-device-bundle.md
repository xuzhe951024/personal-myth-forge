```python
def _device_action_bundle(
    *,
    final_showcase_readiness: dict[str, Any],
    ios_device_evidence_bundle: dict[str, Any],
) -> dict[str, Any]:
    source_report = "final_showcase_readiness"
    source_bundle = final_showcase_readiness.get("device_action_bundle")
    if not isinstance(source_bundle, dict):
        source_report = "ios_device_evidence_bundle"
        source_bundle = ios_device_evidence_bundle.get("device_action_bundle")
    if not isinstance(source_bundle, dict):
        return _missing_device_action_bundle()
    actions = [
        _device_action(action)
        for action in source_bundle.get("actions", [])
        if isinstance(action, dict)
    ]
    first_action = _device_first_action(source_bundle.get("first_action"), actions)
    return {
        "id": "final_demo_launch_device_actions",
        "label": "Final Demo Launch Device Actions",
        "source_report": source_report,
        "status": _normalized_status(str(source_bundle.get("status", "blocked"))),
        "actions": actions,
        "first_action": first_action,
        "summary": _device_action_summary(actions),
        "safety": _device_action_safety(),
    }
```

Add `_device_action`, `_device_next_action`, `_device_first_action`, `_device_action_summary`, `_device_action_safety`, and `_missing_device_action_bundle`. They should copy only scalar fields and preserve `validation_command` when present.
